342  def _term_kill(self):
343  """Issue signals to entire process group. First give processes a
344  chance to exit cleanly with CONT+TERM, then attempt to KILL after
345  a delay."""
346  _log.debug('%s _term_kill', self.log_prefix)
347  os.killpg(self._pgid, signal.SIGCONT)
348  os.killpg(self._pgid, signal.SIGTERM)
349  time.sleep(KILL_WAIT)
350  try:
351  os.killpg(self._pgid, signal.SIGKILL)
352  except ProcessLookupError:
353  # this happens if all processes in the pgroup have already
354  # exited and the group no longer exists, which is what should
355  # happen in most cases
356  pass

358  def _pgroup_wait(self):
359  """Wait until the process group lead by this run no longer exists.
360  Assumes that it should already be exiting normally (e.g. the parent
361  has already exited). If WAIT_DELAY_KILL is reached in expontential
362  back off and the group still exists, SIGKILL is sent to the group.
363  If WAIT_DELAY_GIVE_UP is reached, an error is logged and the function
364  will return. Inspired by proctrack_pgid plugin from slurm."""
365  _log.debug('%s _pgroup_wait max delay %d'
366  % (self.log_prefix, WAIT_DELAY_GIVE_UP))
367  delay = 1
368  signum = 0 # 0 is the null signal, does error checking only
369  while True:
370  try:
371  os.killpg(self._pgid, signum)
372  except ProcessLookupError:
373  # pgroup no longer exists, we are done waiting
374  break
375  # else pgroup still exists
376  time.sleep(delay)
377  delay *= 2
378  if delay > WAIT_DELAY_KILL:
379  signum = signal.SIGKILL
380  _log.warn(
381  '%s pgroup still exists, sending KILL, next delay=%d',
382  self.log_prefix, delay)
383  if delay > WAIT_DELAY_GIVE_UP:
384  _log.error('%s pgroup did not exit', self.log_prefix)
385  break
