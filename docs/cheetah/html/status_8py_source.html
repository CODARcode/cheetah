126 def _print_run_code_output(run_name, run_dir, filter_code=None):

152  continue
153  fpath = outputs[code][k]
154  size = os.path.getsize(fpath)
155  print('>>>', run_name, code, 'std' + k, '(%d bytes)' % size)
156  # TODO: Encountering non utf-8 chars is unlikely,
