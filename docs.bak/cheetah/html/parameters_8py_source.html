101 class Instance(object):
102  """
103  Represent an instance of an application with fixed parameters. An
104  application may consistent of multiple codes running at the same time,
105  and multiple middlewear layers (scheduler like PBS, runner like aprun,
106  or swift), all of which may have their own parameters.
107 
108  Abstractly, an instance is a two-level nested dict, where the first
109  level indicates the target for a parameter (application code or
110  middlewear), and the second level contains the parameter values for that
111  target.
112  """
113  def __init__(self):
114  # abstract container with all param values in a hierarchy based on
115  # their target
116  self._parameter_values = defaultdict(dict)
117 
118  # temporary containers for staging params so derived values can
119  # be calculated after all are added.
120  self._simple_pv_list = defaultdict(list)
121  self._derived_pv_list = defaultdict(list)
122 
123  # subset of paramaters related to application codes that will
124  # need to be run
125  self._code_commands = dict()
126 
127  self._values_calculated = False
128 
129  def add_parameter(self, p, idx):
130  if self._values_calculated:
131  raise ValueError("new parameters can't be added after get")
132  pv = ParameterValue(p, idx)
133  if callable(pv.value):
134  self._derived_pv_list[pv.target].append(pv)
135  else:
136  self._simple_pv_list[pv.target].append(pv)
137 
138  @property
139  def parameter_values(self):
140  """Wrapper to allow delayed calculation of derived parameter values."""
141  if not self._values_calculated:
142  self._calculate_values()
143  return self._parameter_values
144 
145  @property
146  def code_commands(self):
147  """Wrapper to allow delayed calculation of derived parameter values."""
148  if not self._values_calculated:
149  self._calculate_values()
150  return self._code_commands
151 
152  def _calculate_values(self):
153  # Do in two steps, to support derived params across codes / run
154  # components. First step builds a two level dict with top level
155  # keys being target/code name, next level being simple param
156  # values for that target.
157  simple_value_map = {} # passed to fn for derived params
158  for target, target_pv_list in self._simple_pv_list.items():
159  # NB: not attempting to support deriving values from other
160  # derived values.
161  simple_value_map[target] = dict((pv.name, pv.value)
162  for pv in target_pv_list)
163 
164  targets = (set(self._simple_pv_list.keys())
165  | set(self._derived_pv_list.keys()))
166  for target in targets:
167  target_p = self._parameter_values[target]
168 
169  target_pv_list = self._simple_pv_list[target]
170 
171  for derived_pv in self._derived_pv_list[target]:
172  derived_pv.value = derived_pv.value(simple_value_map)
173  target_pv_list.append(derived_pv)
174 
175  for pv in target_pv_list:
176  # Add a command for any code that has at least one param
177  # of any type, even if no command line args or opts.
178  if target not in self._code_commands:
179  self._code_commands[target] = CodeCommand(target)
180 
181  # Custom handling for command line param types
182  if pv.is_type(ParamCmdLineArg):
183  self._code_commands[target].add_arg(pv.position, pv.value)
184  elif pv.is_type(ParamCmdLineOption):
185  self._code_commands[target].add_option(pv.option, pv.value)
186  if pv.name in target_p:
187  raise ValueError('parameter name conflict: "%s"' % pv.name)
188  # Always save the value, regardless of param type.
189  target_p[pv.name] = pv
190  self._values_calculated = True

299 class Param(object):
300  """Abstract base class representing a parameter to an application. This
301  includes any method for modifying the run characteristics of an
302  application - command line, config file, environment variables, different
303  executable built with diffrent compiler flags.
304 
305  Every parameter must have a unique name, and must target a specific
306  application or middleware, e.g. pbs, aprun, or one of the science
307  codes that make up an application.
308 
309  Note that if a science application has only one code, it will likely still
310  involve middlewhere targets like PBS. Using a different target is one way
311  to model those.
312 
313  TODO: is it useful to separate the definition of a param and it's values?
314 
315  TODO: should we require that the name be unique across all targets, or
316  just within each target? Global uniqueness allows for a simple list of
317  dict representation of instances, but two level nested dicts may be
318  more powerful (first level is target, second level is params)."""
319 
320  def __init__(self, target, name, values):
321  self.target = target
322  self.name = name
323 
324  # allow a function to be specified instead of list of values, for
325  # derived parameters. Wrap in a list to simplify the cross
326  # product calculation, calculate during generation when the
327  # other values are known.
328  if callable(values):
329  values = [values]
330 
331  self.values = values
332  assert type(values) == list, "Parameter value must be a list"
333 
334  def __get__(self, idx):
335  return self.values[idx]
336 
337  def __len__(self):
338  return len(self.values)
